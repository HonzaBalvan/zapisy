\begin{definice}
    Mocninnou Еҷadou rozumГӯme $\sum_{k=0}^{\infty}c_k\cdot(z-z_0)^k$, kde $c_k \in \mathbb C$ jsou koeficienty Еҷady, $z_0 \in \mathbb C$ je stЕҷed Еҷady a $z \in \mathbb C$ je \uv{promДӣnnГЎ}.
\end{definice}

\begin{poznamka}
    Co to je? $\begin{cases}\text{Еҷada s parametrem } z \\ \text{speciГЎlnГӯ typ funkce} \\ \text{zobecnДӣnГӯ polynomu} \end{cases}$
\end{poznamka}

\begin{veta}[PolomДӣr konvergence] Je dГЎna mocninnГЎ Еҷada. Pak $\exists R \in [0, \infty]$ t. Еҫ. $\forall z \in \mathbb C$ platГӯ
\begin{enumerate}[i)]
  \item $|z-z_0|<R \implies$ Еҷada konverguje absolutnДӣ,
  \item $|z-z_0|>R \implies$ Еҷada diverguje.
\end{enumerate}
\end{veta}

\begin{proof}
    pomocnГЎ mnoЕҫina $\mathscr K \coloneq \left\{\rho \geq 0; \text{ posloupnost } \{|c_k|\cdot \rho^k\}\text{ je omezenГЎ}\right\}$

    zЕҷejmДӣ $\mathscr K \neq \emptyset (0 \in \mathscr K)$, poloЕҫme $R\coloneq \sup \mathscr K$
    \begin{enumerate}[{ad} i)]
        \item nechЕҘ $z \in \mathbb C, |z-z_0|<R\implies\exists \rho: \rho >|z-z_0|, \rho \in \mathscr K$

            tj. $\exists C>0$ t. Еҫ. $|c_k|\rho^k\leq C, \forall k$

            lze odhadnout: $|c_k(z-z_0)^k|=|c_k||z-z_0|^k = \underbrace{|c_k|\cdot\rho^k}_{\leq C}\cdot\underbrace{(\frac{|z-z_0|}{\rho})^k}_{\eqcolon \theta^k, \theta <1}\leq C \cdot \rho^k \stackrel{V. 10. 3.}{\implies}$ Еҷada konverguje absolutnДӣ
        \item nechЕҘ $z \in \mathbb C, |z-z_0|>R\implies\underbrace{|z-z_0|}_{\eqcolon \rho}\notin \mathscr K$

            tj. $\{|c_k||z-z_0|^k\}_{k=0}^\infty$ nenГӯ omezenГЎ

            ?? kdyby Еҷada konvergovala, pak (dle V. 10. 1.) $|c_k(z-z_0)|\to 0, k \to \infty$

            $|c_k(z-z_0)^k|=|c_k||z-z_0|^k$ coЕҫ nenГӯ omezenГ© $\implies$ \textsc{spor} \Lightning (V. 7. 2. ?))
    \end{enumerate}
\end{proof}

\begin{veta}
    Je dГЎna mocninnГЎ Еҷada. NechЕҘ $c_k \neq 0$, nechЕҘ $|\frac{c_{k+1}}{c_k}|\to r, k \to \infty$. Pak $R \coloneq \frac{1}{r}$ (zde rozumГӯme $\frac{1}{\infty}=0, \frac{1}{0}= \infty$) je polomДӣr konvergence Еҷady.
\end{veta}

\begin{proof}
    \textsc{trik}: uЕҫijeme V. 10. 4. (podГӯl krit.) na Еҷadu $\sum b_k$, kde $_k = |c_k||z-z_0|^k$ (BГҡNO $z\neq z_0$)

    $\frac{b_{k+1}}{b_k}=\frac{|c_{k+1}|}{|c_k|}\cdot \frac{|z-z_0|^{k+1}}{|z-z_0|^k}= \frac{|c_{k+1}|}{|c_k|}\cdot |z-z_0| \to r \cdot |z-z_0|$

      diskuze: \begin{enumerate}[1.]
     \item $\underline{r \in (0, \infty)}: \stackrel{V. 10. 4.}{\implies} \sum b_k$ konverguje, pokud $r\cdot|z-z_0|<1 \implies |z-z_0| < \frac{1}{r}$; $\sum b_k$ diverguje, pokud $r\cdot|z-z_0|>1 \implies |z-z_0| > \frac{1}{r}$

     $\implies$ mocninnГЎ Еҷada $\begin{cases}
       \text{konverguje absolutnДӣ pro } |z-z_0| <\frac{1}{r},& \\\text{urДҚitДӣ diverguje pro } |z-z_0| >\frac{1}{r},&
       \end{cases}$
     \item $\underline{r=0}: \stackrel{V. 10. 4.}{\implies} \sum b_k$ konverguje $\forall z \in \mathbb C\implies R = \infty$,
     \item $\underline{r= \infty}: b_k \cancel{\to} 0, \forall z \in \mathbb C, z \neq z_0 \implies R = 0.$
   \end{enumerate}
\end{proof}

\begin{veta}
NechЕҘ Еҷada $\sum_{k=0}^{\infty}c_k(z - z_0)^k$ mГЎ polomДӣr konvergence $R > 0$. OznaДҚme
\[
F(z) = \sum_{k=0}^{\infty} c_k (z - z_0)^k, \quad f(z) = \sum_{k=1}^{\infty} k c_k (z - z_0)^{k-1}.
\]
Potom $F'(z) = f(z)$ pro $\forall z \in U(z_0, R)$, kde $U(z_0, R) = \{z \in \C; |z - z_0| < R\}$.
\end{veta}

\begin{proof}
  BГҡNO $z_0 = 0, z \in \mathbb C, |z|<R$ pevnГ©

  cГӯl: $\varphi(h)\to 0, h \to 0$, kde $\varphi(h)= \frac{F(z+h)-F(z)}{h}-f(z)$

  navГӯc: $0<|h|<\delta; \delta>0$ pevnГ© t. Еҫ. $|z+\delta|<R$

  $F(z+h)=\cancel{c_0}+\cancel{c_1(z+h)}+ \sum_{k=2}^{\infty} c_k (z+h)^k$

  $F(z) = \cancel{c_0} + \cancel{c_1(z)}+\sum_{k=2}^\infty c_k z^k$

  $f(z) = 0 + \cancel{c_1} + \sum_{k=2}^\infty k\cdot c_k \cdot z_{k-1}$

  $\varphi(h) = \sum_{k=2}^\infty c_k \cdot\frac{1}{h}\cdot [\underbrace{(z+h)^k}_{\mathclap{\textstyle =\cancel{z^k}+\cancel{kz^{k-1}h}+\sum_{j=2}^k {k\choose j}z^{k-j}h^j}}-z^k]-kz^{k-1} \stackrel{\text{binom. v.}}{=} \sum_{k=2}^\infty h\cdot c_k\cdot \underbrace{\sum_{j=2}^\infty {k\choose j} z^{h-j}\cdot h^{j-2}}_{\eqcolon S_k}$

  ukГЎЕҫeme, Еҫe $\sum_{k=2}\infty |c_k||S_k|<K$ nezГЎvisle na $|h|<\delta$, pak $|\varphi(h)|\leq|h|\cdot K \to 0$

  odhady $S_k: S_k = \sum_{l=0}^{k-2} \underbrace{{k \choose l+2}}_{= \frac{k!}{(l+2)!(k-l-2)!}= \frac{k(k-1)(k-2)!}{\underbrace{(l+2)(l+1)}_{\geq 1}l! (k-l-2)!}}$
\end{proof}

\begin{poznamka}
\begin{itemize}
    \item funkce $F(z)$, $f(z)$ jsou pro $z \in U(z_0, R)$ korektnДӣ definovГЎny, neboЕҘ danГ© Еҷady konvergujГӯ absolutnДӣ (Lemma 11.1.)
    \item tvrzenГӯ platГӯ ve smyslu derivace podle komplexnГӯ promДӣnnГ©, tj.
    \[
    (\forall \varepsilon > 0)(\exists \delta > 0)\left[0 < |h| < \delta, h \in \C \Longrightarrow \left| \frac{F(z+h) - F(z)}{h} - f(z) \right| < \varepsilon \right],
    \]
    kde $z \in U(z_0, R)$ je libovolnГ©.
    \item speciГЎlnГӯ pЕҷГӯpad je i derivace podle reГЎlnГ© promДӣnnГ©, tj. $F'(x) = f(x)$, pro kaЕҫdГ© $x$ z intervalu $(-R, R)$
\end{itemize}
\end{poznamka}

\begin{dusledek}
  Na mnoЕҫinДӣ $U(z_0, R)$ platГӯ:
\begin{itemize}
    \item funkce $F(z)$ je nekoneДҚnДӣ diferencovatelnГЎ a $F''(z) = \sum_{k=2}^{\infty} k(k-1)c_k(z - z_0)^{k-2}$, $F^{(3)}(z) = \sum_{k=3}^{\infty} k(k-1)(k-2)c_k(z - z_0)^{k-3}$ atd.
    \item funkce $\Phi(z) = \sum_{k=0}^{\infty} \frac{c_k}{k+1}(z - z_0)^{k+1}$ je primitivnГӯ funkce k $F(z)$, tj. $\Phi'(z) = F(z)$
\end{itemize}
\end{dusledek}

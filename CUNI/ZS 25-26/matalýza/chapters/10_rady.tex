\begin{definice}
  NechЕҘ $a_k \in \mathbb R$ pro $k \in \mathbb N \,(\mathbb N_0)$. Symbol $\sum_{k=1}^{\infty} a_k$ se nazve Еҷada. Posloupnost $\{s_n\}_{n=1}^{\infty}$, kde $s_n \coloneq \sum_{k=1}^{n} a_k$ nazveme posloupnost ДҚГЎsteДҚnГҪch souДҚtЕҜ Еҷady. ДҢГӯslo $s \coloneq \lim_{n \to \infty} s_n \in \mathbb R^*$ se nazve souДҚet Еҷady. Pokud $s \in \mathbb R$ Еҷekneme, Еҫe Еҷada konverguje, v opaДҚnГ©m pЕҷГӯpadДӣ diverguje. Pokud $s = \pm \infty$ Еҷada diverguje do $\pm \infty$ (urДҚitДӣ diverguje), pokud $\nexists \lim_{n\to \infty} s_n$ Еҷada osciluje (diverguje).
\end{definice}

\begin{priklad}
  \begin{enumerate}[1)]
    \item $\sum_{k = 1}^{\infty} \frac{1}{k(k+1)} = 1$, neboЕҘ \textsc{trik} $a_k = \frac{1}{k}-\frac{1}{k+1}$, takЕҫe $s_n = (1- \frac{1}{2})+(\frac{1}{2}-\frac{1}{3})+\dots = 1 - \frac{1}{n} \to 1, n \to \infty$
    \item $\sum_{k=0}^{\infty} q^k = \frac{1}{1-q}, \forall q \in (-1,1)$

\begin{proof}
  $s_n = 1+q+q^2+\dots+q^n = \frac{1-q^{n-1}}{1-q} \to \frac{1}{1-q}, n \to \infty$
\end{proof}

    \item $\sum_{k=1}^{\infty} \frac{1}{k} = \infty$, viz vГҪЕЎe
    \item $\sum_{k=1}^{\infty} (-1)^{k}$ osciluje (mГЎ dva hromadnГ© body $\implies$ nemГЎ limitu)
  \end{enumerate}
\end{priklad}

\begin{veta}[10.1. NutnГЎ podmГӯnka konvergence Еҷady]
  NechЕҘ $\sum_{k=1}^{\infty} a_k$ konverguje. Pak $a_k \to 0, k \to \infty$.
\end{veta}

\begin{proof}
  posuzuji: $a_n = s_n  s_{n-1}, \forall n \geq 2$

  vГӯme: $s_n \to s \in \mathbb R$, a tedy takГ© $s_{n-1} \to s \implies a_n \to s-s = 0$ (VoAL kon.)
\end{proof}

\begin{lemma}[L.10.1]
  NechЕҘ $a_k = b_k, \forall k \in \mathbb N$ aЕҫ na koneДҚnДӣ mnoho vГҪjimek. Pak $\sum_{k=1}^{\infty} a_k$ konverguje $\iff \sum_{k=1}^{\infty} b_k$ konverguje.
\end{lemma}

\begin{proof}
  $\exists n_0 \in \mathbb N$ t. Еҫ. $a_k = b_k, \forall k \geq n_0$. $s_n = \sum_{k=1}^{n} a_k = a_1+a_2+\dots+a_{n_0}+a_{n_0+1}+\dots+a_n, t_n = \sum_{k=1}^{n} b_k = b_1+b_2+\dots+b_{n_0}+b_{n_0+1}+\dots+b_n$ -- od ДҚlenЕҜ $a_{n_0}, b_{n_0}$ se sДҚГӯtance rovnajГӯ $\implies \exists c \in \mathbb R$ t. Еҫ. $s_n-t_n=c, \forall n \geq n_0$, a tedy $s_n \to s \in \mathbb R \iff t_n \to t \in \mathbb R, t = s-c$
\end{proof}

\begin{poznamka}
  ДҢasto pГӯЕЎeme $\sum a_k$ mГӯsto $\sum_k a_k$ nebo $\sum_{k=1}^{\infty} a_k$.
\end{poznamka}

\begin{lemma}[L. 10. 2.]
  NechЕҘ $a_k \geq 0$. Potom Еҷada $\sum_{k=1}^{\infty} a_k$ konverguje $\iff$ mГЎ omezenГ© ДҚГЎsteДҚnГ© souДҚty.
\end{lemma}

\begin{proof}
  $s_n \coloneq \sum_{k=1}^{n} a_k$

  $s_{n+1} = s_n + a_{k+1}$, pЕҷiДҚemЕҫ $a_{k+1} \geq 0$, tedy $s_{n+1} \geq s_n$

  Tedy $\{s_n\}_{n=1}^{\infty}$ je neklesajГӯcГӯ, potom tedy dle V. 7. 2. konverguje $\iff$ je omezenГЎ.
\end{proof}

\begin{veta}[V. 10. 3. SrovnГЎvacГӯ kritГ©rium -- nelimitnГӯ verze]
  Jsou dГЎny Еҷady $\sum_{k=1}^{\infty} a_k$ a $\sum_{k=1}^{\infty} b_k$, kde $a_k, b_k \geq 0$. NechЕҘ existuje $c>0$ a $n_0$ t. Еҫ. $a_k \leq c\cdot b_k, \forall k \geq n_0$. Pak

  \begin{enumerate}[i)]
    \item pokud Еҷada $\sum_{k=1}^\infty b_k$ konverguje $\implies$ Еҷada $\sum_{k=1}^\infty a_k$ konverguje,
    \item pokud Еҷada $\sum_{k=1}^\infty a_k$ diverguje $\implies$ Еҷada $\sum_{k=1}^\infty b_k$ diverguje.
  \end{enumerate}
\end{veta}

\begin{proof}
  Z L. 10. 1. mЕҜЕҫeme pЕҷedpoklГЎdat, Еҫe $\forall k \in \mathbb N: a_k \leq c \cdot b_k$ (mЕҜЕҫu opomenout koneДҚnДӣ mnoho ДҚlenЕҜ).

  $s_n \coloneq \sum_{k=1}^{\infty} a_k, t_n \coloneq \sum_{k=1}^{\infty} b_k$

  $s_n = \sum a_k \leq \sum c \cdot b_k = c \cdot t_n$

  UЕҫijme L.10.2.:

  \begin{enumerate}[i)]
    \item $\sum b_k$ konverguje $\implies \{t_n\}_{n=1}^{\infty}$ je omezenГЎ $\implies \{c \cdot t_n\}_{n=1}^{\infty}$ je omezenГЎ $\implies \{s_n\}_{n=1}^{\infty}$ je omezenГЎ $\implies \sum a_k$ konverguje
    \item obdobnДӣ
  \end{enumerate}
\end{proof}

\begin{veta}[V. 10. 4. PodГӯlovГ© kritГ©rium]
  Je dГЎna Еҷada $\sum_{k=1}^{\infty} a_k$. NechЕҘ $a_k >0$ a $\frac{a_k+1}{a_k} \to q, k \to \infty$. Pak

  \begin{enumerate}[i)]
    \item je-li $q<1 \implies \sum a_k$ konverguje,
    \item je-li $q>1 \implies \sum a_k$ diverguje.
  \end{enumerate}
\end{veta}

\begin{proof}
  \begin{enumerate}[i)]
    \item \underline{$q<1$}: $\exists \varepsilon > 0$ t. Еҫ. $\theta = q+\varepsilon<1$

    $\lim_{k\to\infty} \frac{a_{k+1}}{a_k} = q \implies q \in \mathscr U(q, \varepsilon) \implies \exists n_0 \in \mathbb N: \forall k \geq n_0: \frac{a_{k+1}}{a_k}<\theta$.

    Tedy $a_{k+1} \leq \theta \cdot a_k$.

    Dle L. 10. 1. BГҡNO $\forall k \in \mathbb N:a_{k+1} \leq \theta \cdot a_k$

    $\implies a_{k+1} \leq \theta\cdot a_k \leq \theta^2 a_{k-1} \leq \dots \theta^k a_1 \implies a_k \leq \theta^{k-1} \cdot a_1$

    $c \coloneq \frac{a_1}{\theta} \implies a_k \leq c \cdot \theta^k$ -- uЕҫijeme V. 10. 3. i)

    $\{\theta_k\}$ geometrickГЎ Еҷada s kvocientem s velikostГӯ menЕЎГӯ neЕҫ 1 $\implies \sum \theta_k$ konverguje $\implies \sum a_k$ konverguje
    \item \underline{$q>1$}: $\exists n_0 \in \mathbb N: \forall k \geq n_0: \frac{a_{k+1}}{a_k} > 1$.

    Pak $a_{k+1} >a_k >0, \forall k \geq n_0 \implies \sum a_k \geq \sum a_{n_0} = \infty \implies \sum a_k$ diverguje.
  \end{enumerate}
\end{proof}


\begin{priklad}
  \begin{enumerate}[1)]
    \item $\sum \frac{1}{k!}$

    $a_k >0, \forall k \in \mathbb N, \lim_{k\to \infty} \frac{a_{k+1}}{a_k} = \lim_{k\to \infty} \frac{k!}{(k+1)!} = \lim_{k\to\infty} \frac{1}{k} = 0 = q$

    $\implies \sum \frac{1}{k!}$ konverguje
    \item $\sum \frac{k^2}{2^k}$

    $a_k >0, \forall k \in \mathbb N, \lim_{k \to \infty} \frac{a_{k+1}}{a_k} = \lim_{k\to \infty} \frac{k^2+2k+1}{2k^2} = \frac{1}{2} \implies$ konverguje
  \end{enumerate}
\end{priklad}

\begin{poznamka}
  Pokud $\frac{a_{k+1}}{a_k} \to 1, k \to \infty$ nelze obecnДӣ rozhodnout. NapЕҷ. $\sum \frac{1}{k}$ diverguje, ale $\sum \frac{1}{k^2}$ konverguje.
\end{poznamka}

\begin{veta}[V. 10. 6. IntegrГЎlovГ© kritГ©rium]
  NechЕҘ $\sum a_k$ Еҷada, $a_k > 0$. NechЕҘ existuje spojitГЎ nezГЎpornГЎ nerostoucГӯ funkce $f: [1, \infty] \to \mathbb R$ t. Еҫ. $a_k = f(k)$. Pak $\sum a_k$ konverguje $\iff (\mathscr N) \int_1^\infty f(x)dx <\infty$.
\end{veta}
OBRAZEKOBRAZEk

\begin{proof}
  $s_n \coloneq \sum^n a_k$. VГӯme, Еҫe $\sum a_k$ konverguje $\iff \{s_n\}$ je omezenГЎ (shora). Dle V. 9. 6. mГЎ $f$ primitivnГӯ funkci, nazvДӣme $F$, pak $(\mathscr N)\int f(x)dx = [F(x)]_1^\infty = \lim_{x \to \infty} F(x) - \lim_{x \to 1^+} F(x)$, tedy $(\mathscr N)\int f(x) dx < \infty \iff \lim_{x \to \infty} F(x) \neq \infty$.

  Je-li $k \in \mathbb N$ pevnГ©, $x \in [k, k+1]$, pak $f(k) \geq f(x) \geq f(k+1)$.

  $a_k \geq f(x) \geq a_{k+1}$

  $\int_k^{k+1} a_k dx \geq \int_k^{k+1} f(x) dx \geq \int_k^{k+1} a_{k+1}$

  $a_k \geq \int_k^{k+1} f(x) dx \geq a_{k+1}$

  $\sum_{k=1}^n a_k \geq \sum_{k=1}^n \int_k^{k+1} f(x) dx \geq \sum_{k=1}^n a_{k+1}$

  $s_n \geq \int_1^{n+1} f(x) dx \geq s_{n+1} -a_1$

  pro $ n \to \infty: \lim_{n \to \infty} s_n \geq \int_1^\infty f(x) dx \geq \lim_{n \to \infty} s_{n+1}-a_1$ -- kaЕҫdГЎ strana ekvivalence vyplГҪvГЎ z jednГ© nerovnice
\end{proof}

\begin{veta}[V. 10. 7. SrpvnГЎvacГӯ kritГ©rium -- limitnГӯ verze]
  NechЕҘ $a_k, b_k >0$, nechЕҘ $a_k \sim b_k, k \to \infty$. Pak $\sum a_k$ konverguje $\iff \sum b_k$ konverguje.
\end{veta}

\begin{proof}
  VГӯme, Еҫe $\frac{a_k}{b_k} \to c \neq 0$, nutnДӣ $c >0$.
  Volme $\varepsilon >0$ t. Еҫ. $0 \in \mathscr U(c, \varepsilon)$.
  $\implies \exists n_0 \in \mathbb N: \frac{a_k}{b_k} \in \mathscr U(c, \varepsilon), \forall k \geq n_0$
  $\implies c- \varepsilon < \frac{a_k}{b_k} < c + \varepsilon$
  $c_1 \coloneq c - \varepsilon, c_2 \coloneq c+\varepsilon$
  $0 < c_1 \cdot b_k < a_k < c_2 \cdot b_k, \forall k \geq n_0$ (dle L. 10. 1. BГҡNO $\forall k \in \mathbb N$) -- mГЎm 2x V. 10. 3.
\end{proof}
